#include <bits/stdc++.h>
using namespace std;
using ll = long long;

void solve(){
    int n, k, m;
    cin >> n >> k >> m;
    vector <int> a(n+1), cnt(k+1), Pre(n+2);
    for (int i = 1; i <= n; ++ i) cin >> a[i];
    for (int i = 1; i <= n; ++ i)
        Pre[i] = cnt[a[i]], ++ cnt[a[i]];
    
    fill(cnt.begin(), cnt.end(), 0);
    vector <int> ans;

    int round = m/n;
    for (int i = n; i >= 1; -- i){
        int sum = Pre[i] + cnt[a[i]], res = 0;
        int r;
        if (round*n != m) r = round*n+i <= m ? round+1 : round;
        else r = round;
        if (Pre[i]%2){  //前面有奇数个
            res += (r+1)/2;  //加上奇数回合
            if (sum%2) res += r/2;  //加上偶数回合
        }
        else {  //前面有偶数个
            if (sum%2) res += 0;
            else res += r/2;  //后面有偶数个，加上偶数回合
        }
        ans.push_back(res);
        ++ cnt[a[i]];
    }
    reverse(ans.begin(), ans.end());
    for (int i = 0; i < ans.size(); ++ i){
        cout << ans[i];
        if (i != ans.size()-1) cout << " ";
    }
    cout << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
