#include <bits/stdc++.h>
using namespace std;
using ll = long long;
string dx[] = {"chimasu", "rimasu", "mimasu", "bimasu", "nimasu", "kimasu", "gimasu", "shimasu", "ikimasu"};
string dy[] = {"tte", "tte", "nde", "nde", "nde", "ite", "ide", "shite", "itte"};

void solve(){
    string s;
    cin >> s;
    if (s == dx[8]){
        cout << dy[8] << "\n";
        return;
    }
    for (int i = 0; i < s.size(); ++ i){
        for (int j = 0; j < 8; ++ j)
            if (s.substr(i) == dx[j]){
                cout << s.substr(0, i) << dy[j] << "\n";
                return;
            }
    }
    return;
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
