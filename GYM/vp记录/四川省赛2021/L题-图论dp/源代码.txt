#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int N = 1e5+11, inf = 0x3f3f3f3f;
int f[N][105], w[N];  //f[i][j]表示第i个点到辣度为j的最短距离
vector <int> g[N];
int n, m, q;

void bfs(int k){
    queue <int> q;
    for (int i = 1; i <= n; ++ i)
        if (w[i] <= k){
            f[i][k] = 0;
            q.push(i);
        }
    while (q.size()){
        auto u = q.front();
        q.pop();
        for (auto &x : g[u]){
            if (f[x][k] > f[u][k]+1)
                q.push(x), f[x][k] = f[u][k] + 1;
        }
    }
}

void solve(){
    memset(f, 0x3f, sizeof f);
    cin >> n >> m >> q;
    for (int i = 1; i <= n; ++ i) cin >> w[i];
    while (m--){
        int a, b;
        cin >> a >> b;
        g[a].push_back(b), g[b].push_back(a);
    }

    for (int i = 1; i <= 100; ++ i) bfs(i);

    while (q--){
        int u, weight;
        cin >> u >> weight;
        cout << (f[u][weight] == inf ? -1 : f[u][weight]) << "\n";
    }
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
