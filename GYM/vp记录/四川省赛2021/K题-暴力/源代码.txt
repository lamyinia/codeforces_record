#include <bits/stdc++.h>
using namespace std;
using ll = long long;

void solve(){
    int n, k;
    cin >> n >> k;
    if (k >= n){
        for (int i = 1; i <= n; ++ i) cout << i << " ";
        cout << "\n";
        return;
    }

    vector <int> p, vit(n+1);
    int x = 1;
    while (x <= n){
        while (vit[x]) ++ x;
        int u = x;
        while (u <= n) p.push_back(u), vit[u] = true, u += k;
    }
    for (auto &it : p) cout << it << " ";
    cout << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
