template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }

}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int n;
    cin >> n;
    map<array<int,2>, int> mps;

    i64 res = 0, c1 = 0, c2 = 0;
    for (int i = 0, x, y, u, v; i < n; ++ i){
        cin >> x >> y >> u >> v;
        int dx = u-x, dy = v-y, d = abs(__gcd(dx, dy));
        dx /= d, dy /= d;
        int ux = -dx, uy = -dy;
        res += mps[{uy, ux}];
        mps[{dy, dx}] += 1;
    }

    cout << res << "\n";
}
