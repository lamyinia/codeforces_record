        cout << "? " << l << " " << r << "\n" << flush;

        return x;
    };

    int l = 1, r = 999;
    while (l < r){
        int ml = l + (r-l)/3, mr = r - (r-l)/3;
        int sum = ops(ml, mr);
        if (sum == ml * mr) l = mr+1;
        else if (sum == ml * (mr+1)) l = ml+1, r = mr;
        else r = ml;
    }

    cout << "! " << l << "\n" << flush;
}
