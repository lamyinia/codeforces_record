const int N = 5050, mod = 1e9 + 7;

/*
    考虑到让字典序尽可能大，先按价值排序
    让后尽可能选更多次，如果选的次数相同，则从价值更大且位置更靠右的选
*/
void solve(){
    int n, k;
    cin >> n;
    vector <int> ans(n+2), nums(n+1);
    for (int i = 1; i <= n; ++ i) cin >> nums[i];
    cin >> k;
    int mins = 2e9;
    for (int i = n; i >= 1; -- i){
        mins = min(mins, nums[i]);
        nums[i] = mins;
    }
    ans[0] = 2e9;
    for (int i = 1; i <= n; ++ i){
        if (nums[i] == nums[i-1]) ans[i] = ans[i-1];
        else ans[i] = min(k/(nums[i]-nums[i-1]), ans[i-1]);
        k -= (nums[i]-nums[i-1]) * ans[i];
    }

    for (int i = 1; i <= n; ++ i) cout << ans[i] << "\n "[i < n];
}

int main(){
#ifdef ONLINE_JUDGE

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--)
        solve();
    return 0;
}
