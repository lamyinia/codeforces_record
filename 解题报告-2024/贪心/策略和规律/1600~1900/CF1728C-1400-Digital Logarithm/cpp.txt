#include <bits/stdc++.h>
using namespace std;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    priority_queue <int> pq1, pq2;
    for (int i = 0,x; i < n; ++ i){
        cin >> x;
        pq1.push(x);
    }
    for (int i = 0,x; i < n; ++ i){
        cin >> x;
        pq2.push(x);
    }
    auto find = [&](int x)->int{
        int res = 0;
        while (x) x /= 10, res += 1;
        return res;
    };
    int res = 0;
    while (!pq1.empty() && !pq2.empty()){
        int x = pq1.top(), y = pq2.top();
        if (x > y){
            pq1.pop();
            pq1.push(find(x));
            res += 1;
        } else if (x < y){
            pq2.pop();
            pq2.push(find(y));
            res += 1;
        } else {
            pq1.pop();
            pq2.pop();
        }
    }

    cout << res << "\n";
}

    int _ = 1;
    cin >> _;
    while (_--) solve();

}
