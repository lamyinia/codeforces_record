#include <bits/stdc++.h>
using namespace std;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector <int> A(n);
    string s;
    cin >> s;
    for (int i = 0; i < n; ++ i){
        A[i] = s[i] - '0';
    }

    int res = 1;

    for (int k = 2; k <= n; ++ k){
        vector <int> tag(n);
        int cur = 0;
        for (int i = 0; i+k-1 < n; ++ i){
            cur ^= tag[i];
            if ((cur^A[i]) != 1){
                cur ^= 1;
                if (i+k < n) tag[i+k] ^= 1;
            }
        }
        bool jg = 1;
        for (int i = n-k+1; i < n; ++ i){
            cur ^= tag[i];
            if ((cur^A[i]) != 1){
                jg = 0;
                break;
            }
        }
        if (jg) fmax(res, k);
    }

    cout << res << "\n";
}


int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
