#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int N = 3e5;
 
void solve(){ 
    string s;
    cin >> s;
    int cnt = 1, key = 0;
    for (int i = 1; i < s.size(); ++ i){
        if (s[i] != s[i-1]) ++ cnt;
        if (s[i] == '1' && s[i-1] == '0') key = 1;
    }
    cout << cnt - key << "\n";
}
