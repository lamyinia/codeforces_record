void solve(){
    int n, m;
    cin >> n >> m;
    vector <int> A(n+1), B(m+1), C;
    for (int i = 1; i <= n; ++ i) cin >> A[i];
    for (int i = 1; i <= m; ++ i) cin >> B[i];
    sort(B.begin()+1, B.end(), greater<int>());
    

    for (int i = 1, j = 1; i <= n; ++ i){
        while (j <= m && A[i] <= B[j]) C.push_back(B[j]), ++ j;
        C.push_back(A[i]);
        if (i == n && j <= m){
            while (j <= m) C.push_back(B[j]), ++ j;
        }
    }

    for (auto &x : C) cout << x << " ";
    cout << "\n";
}
