#include <bits/stdc++.h>
using namespace std;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}
/*
	设v(n)是不超过n的最大素数，u(n)是大于n的最小素数，求 sigma_2^n {1/(u(i)v(i))}
*/
void solve(){
    auto chk = [&](int x)->bool{
        for (long long i = 2; i*i <= x; ++ i){
            if (x%i == 0){
                return false;
            }
        }
        return true;
    };

    int n;
    cin >> n;
    int a = n, b = (n+1);
    while (!chk(a)) a -= 1;  // 不超过n的最大素数
    while (!chk(b)) b += 1;  //大于n的最小素数
    
    long long son = 1ll*a*b + 2*n + 2 - 2ll*(a+b);
    long long mom = 2ll*a*b;
    long long d = __gcd(son, mom);
    son /= d, mom /= d;
    cout << son << "/" << mom << "\n";
}

    int _ = 1;
    cin >> _;
    while (_--) solve();

}
