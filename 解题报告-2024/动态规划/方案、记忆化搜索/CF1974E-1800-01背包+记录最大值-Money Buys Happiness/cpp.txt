#include <bits/stdc++.h>
using namespace std;
using ll = long long;

void solve(){
    int n, x;
    cin >> n >> x;
    int h_max = 0;
    vector <int> h(n+1), cost(n+1);
    for (int i = 1; i <= n; ++ i){
        cin >> cost[i] >> h[i];
        h_max = max(h_max, h[i]);
    }
    vector <vector<int>> f(n+1, vector<int>(n*h_max+1));
    vector <vector<long long>> g(n+1, vector<long long>(n*h_max+1));

    for (int i = 0; i < n; ++ i){
        f[i][0] = 1;
        if (i > 0){
            for (int j = 0; j <= i * h_max; ++ j){
                if (f[i][j]){
                    g[i][j] += x;
                }
            }
        }
        for (int j = 0; j <= i * h_max; ++ j){
            if (f[i][j]){
                f[i+1][j] = 1;
                g[i+1][j] = g[i][j];
            }
        }
        for (int j = 0; j <= i * h_max; ++ j){
            if (f[i][j] && g[i][j] - cost[i+1] >= 0){
                int js = j + h[i+1];
                if (f[i+1][js]){
                    g[i+1][js] = max(g[i+1][js], g[i][j] - cost[i+1]);
                } else {
                    f[i+1][js] = 1;
                    g[i+1][js] = g[i][j] - cost[i+1];
                }
            }
        }
    }

    int res = 0;
    for (int i = 1; i <= n * h_max; ++ i){
        if (f[n][i]) res = i;
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
