#include <bits/stdc++.h>
using namespace std;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n, c;
    cin >> n >> c;
    vector <int> w(n);
    for (auto &x : w) cin >> x;
    vector <vector<int>> g(n);
    vector <vector<long long>> f(n, vector<long long>(2));
    for (int i = 0,x,y; i < n-1; ++ i){
        cin >> x >> y;
        x -= 1;
        y -= 1;
        g[x].push_back(y);
        g[y].push_back(x);
    }

    auto dfs = [&](auto &&self, int x, int fa)->void{
        f[x][1] = w[x];
        for (auto &y : g[x]){
            if (y == fa) continue;
            self(self, y, x);

            f[x][0] += max(f[y][0], f[y][1]);
            f[x][1] += max(f[y][0], f[y][1]-2*c);
        }
    };

    dfs(dfs, 0, -1);

    cout << max(f[0][0], f[0][1]) << "\n";
}


int main(){
#ifdef ONLINE_JUDGE

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
