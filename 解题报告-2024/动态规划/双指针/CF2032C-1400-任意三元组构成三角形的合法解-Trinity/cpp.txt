#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector <int> A(n);
    for (auto &x : A) cin >> x;

    sort(A.begin(), A.end());

    int res = n-2;
    for (int i = 0, j = -1; i < n-2; ++ i){
        while (j+1 < n && A[i]+A[i+1] > A[j+1]){
            j += 1;
        }
        res = min(res, n - (j-i+1));
    }
    
    cout << res << "\n";
}

    int _ = 1;
    cin >> _;
    while (_--) solve();

}
