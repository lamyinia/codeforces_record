#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}

void solve(){
    const int mod = 1e9+7;

    string s;
    int k;
    cin >> s >> k;
    if (k == 0){
        cout << 1 << "\n";
        return;
    }

    vector <int> ops(1<<10);
    for (int i = 2; i < 1<<10; ++ i){
        ops[i] = ops[__builtin_popcount(i)] + 1;
    }

    vector <int> f(1<<10);
    int cur = 0;
    for (auto &c : s){
        for (int j = (1<<10)-1; j > 0; -- j){
            f[j] += f[j-1];
            f[j] %= mod;
        }
        if (c == '1'){
            f[cur] += 1;
            f[cur] %= mod;
            cur += 1;
        }
    }

    int res = 0;
    for (int i = 1; i < 1<<10; ++ i){
        if (ops[i] == k-1){
            res += f[i];
            res %= mod;
        }
    }
    
    if (ops[cur] == k-1){
        res += 1;
        if (res >= mod) res -= mod;
    }
    if (k == 1) -- res;
    
    cout << res << "\n";
}


int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(nullptr), cout.tie(nullptr);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
