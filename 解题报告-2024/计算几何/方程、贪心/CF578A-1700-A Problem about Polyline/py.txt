import math

def main():
    a, b = map(int, input().split())
    if b > a:
        print(-1)
        return

    ans = math.inf
    k = (a-b) // (2*b)
    if k > 0:
        ans = min(ans, (a-b)/(2*k))
    k = (a+b) // (2*b)
    ans = min(ans, (a+b)/(2*k))

    print(f'{ans:.10f}')

if __name__ == '__main__':
    _ = 1
    while _ != 0:
        _ -= 1
        main()
