#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using ull = unsigned long long;
const int mod = 998244353;

void solve(){
    const ll inf = 1e18;

    ull n;
    cin >> n;
    vector <ull> ans;

    auto f = [&](ull x, int i)->ull{
        if (x-1 > 2*inf/x) return inf+1;
        ull res = x*(x-1)/2;
        if (res > n) return inf+1;
        while (i--){
            res += x;
            if (res > n) return inf+1;
            x *= 2;
        }
        return res;
    };
    auto call = [&](int i)->void{
        #define id(x) (2*x-1)
        ull l = 1, r = inf, bas = 1ll<<i;
        while (l < r){
            ull mid = l+r+1 >> 1;
            if (f(id(mid), i) <= n) l = mid;
            else r = mid-1;
        }
        if (f(id(l), i) == n) ans.push_back(bas*id(l));
    };

    for (int i = 0; i <= 60; ++ i) call(i);

    if (!ans.size()) cout << -1 << "\n";
    else {
        sort(ans.begin(), ans.end());
        for (auto &x : ans) cout << x << "\n";
    }
}
 
int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
