#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int mod = 1e9+7, INF = 1e9, N = 1e5+11;

void solve(){
    int n;
    cin >> n;
    vector <int> nums(n), w(n);
    for (int i = 0; i < n; ++ i) cin >> nums[i], nums[i] -= 1;
	/*
		w[i]表示删去第i个数 产生 w[i] 个贡献
	*/
    for (int i = 0, mx1 = -1, mx2 = -1; i < n; ++ i){
        if (nums[i] > mx1){
            -- w[nums[i]];
            mx2 = mx1, mx1 = nums[i];
        } else if (nums[i] > mx2){
            ++ w[mx1];
            mx2 = nums[i];
        }
    }
    int res = 0;
    for (int i = 0; i < n; ++ i)
        if (w[i] > w[res]) res = i;
    cout << res+1 << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
