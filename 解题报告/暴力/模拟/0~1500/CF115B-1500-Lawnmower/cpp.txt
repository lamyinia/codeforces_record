#include <bits/stdc++.h>
using namespace std;

void solve(){
    int n, m;
    cin >> n >> m;
    
    vector <string> s(n);
    for (int i = 0; i < n; ++ i) cin >> s[i];

    vector <array<int,2>> lrs;
    vector <int> dirs;
    int res = 0;

    for (int i = 0; i < n; ++ i){
        int l = -1, r = -1;
        for (int j = 0; j < m; ++ j){
            if (s[i][j] == 'W'){
                if (l==-1) l = j;
                r = j;
            }
        }
        if (l != -1){
            lrs.push_back({l, r});
            dirs.push_back({i%2});
            res = i;
        }
    }

    for (int i = 0, cur = 0; i < lrs.size(); ++ i){
        int l = lrs[i][0], r = lrs[i][1], dir = dirs[i];
        if (dir == 0){
            if (cur < l) res += r-cur;
            else res += cur-l + r-l;
            cur = r;
        } else {
            if (cur > r) res += cur-l;
            else res += r-cur + r-l;
            cur = l;
        }
    }

    cout << res << "\n";
}
 
int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
