#include <bits/stdc++.h>
using namespace std;
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector <int> nums(n), order;
    for (auto &x : nums) cin >> x;

    for (int i = 1; 1ll*i*i <= n; ++ i){
        if (n%i == 0){
            int j = n/i;
            order.push_back(i);
            if (i != j) order.push_back(j);
        }
    }

    int res = 0;
    for (auto &L : order){
        int g1 = 0;
        for (int i = 0; i < L; ++ i){
            int g2 = 0;
            for (int j = i+L; j < n; j += L){
                g2 = __gcd(g2, abs(nums[j]-nums[i]));
                if (g2 == 1) break;
            }
            g1 = __gcd(g1, g2);
            if (g1 == 1) break;
        }
        res += g1 != 1;
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
