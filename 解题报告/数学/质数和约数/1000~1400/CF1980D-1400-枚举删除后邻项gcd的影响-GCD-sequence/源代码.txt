#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int N = 10005, inf = 1e9;

void solve(){   
    int n;
    cin >> n;
    vector <int> a(n+1), b(n+2), f1(n+1), f2(n+1);
    for (int i = 1; i <= n; ++ i) cin >> a[i];

    function<int(int, int)> gcd = [&](int x, int y){
        return !y ? x : gcd(y, x%y);
    };
    for (int i = 1; i < n; ++ i){
        b[i] = gcd(a[i], a[i+1]);
    }
    f1[0] = 1;
    for (int i = 1; i < n; ++ i)
        f1[i] = f1[i-1] & (b[i] >= b[i-1]);
    f2[n] = 1, b[n] = inf;
    for (int i = n-1; i > 0; -- i)
        f2[i] = f2[i+1] & (b[i] <= b[i+1]);
    
    int f = f1[n-2] | f2[2];
    // 枚举要删除哪一个数字
    for (int i = 2; i <= n-1; ++ i){
        int k = gcd(a[i-1], a[i+1]);
        f |= (k >= b[i-2] && k <= b[i+1] && f1[i-2] && f2[i+1]);
    }
    cout << (f ? "YES\n" : "NO\n");
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
