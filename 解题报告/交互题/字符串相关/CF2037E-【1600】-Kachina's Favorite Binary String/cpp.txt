    if (b < a) a = b;
}

void solve(){
    auto query = [&](int l, int r)->int{
        cout << "? " << l << " " << r << "\n" << flush;
        int ans;
        cin >> ans;
        return ans;
    };

    int n;
    cin >> n;
    vector <int> A(n);
    for (int i = 1; i < n; ++ i){
        A[i] = query(1, i+1);
    }
    if (A == vector<int>(n,0)){
        cout << "! IMPOSSIBLE\n" << endl;
        return;
    }
    int i = 0;
    while (A[i] == 0){
        i += 1;
    }
    string s(n, '0');
    for (int j = 0; j < i-A[i]; ++ j){
        s[j] = '1';
    }
    for (int j =i; j < n; ++ j){
        s[j] = (A[j] > A[j-1] ? '1' : '0');
    }
    cout << "! " << s << endl;
}
