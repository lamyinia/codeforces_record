
#include <bits/stdc++.h>
using namespace std;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector<vector<int>> g(n+3,vector<int>(n+3)), tags(n+3,vector<int>(n+3));

    for (int i = 1; i <= n; ++ i){
        string s;
        cin >> s;
        for (int j = 1; j <= n; ++ j){
            g[i][j] = s[j-1]-'0';
        }
    }

    int res = 0;
    auto call_tag = [&](int x, int y)->void{
        g[x][y] = !g[x][y];
        g[x+1][y] = !g[x+1][y];
    };
    auto push_tag = [&](int x, int y)->void{
        int cnt = tags[x][y];
        if (y != 1) tags[x+1][y-1] += cnt;
        if (y != n) tags[x+1][y+1] += cnt;
        if (y!=1 && y!=n) tags[x+2][y] += cnt;
    };
    auto handle = [&](int x, int y)->void{
        if (tags[x][y]%2) call_tag(x, y);

        if (g[x][y]){
            res += 1;
            tags[x][y] += 1;
            call_tag(x, y);
        }

        if (tags[x][y]%2) push_tag(x, y);
    };

    for (int i = 1; i <= n; ++ i){
        for (int j = 1; j <= n; ++ j){
            handle(i, j);
        }
    }

    cout << res << "\n";
}

    int _ = 1;

}
