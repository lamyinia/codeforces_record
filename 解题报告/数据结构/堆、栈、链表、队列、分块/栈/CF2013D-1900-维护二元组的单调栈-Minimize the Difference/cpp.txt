#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector <ll> A(n+1);
    for (int i = 1; i <= n; ++ i) cin >> A[i];

    auto ceil = [&](ll x, ll y)->ll{
        return (x+y-1) / y;
    };
    vector <array<ll,2>> stk(n);

    int top = -1;
    for (int i = 1; i <= n; ++ i){
        array<ll,2> cur = {A[i], 1};
        while (top >= 0 && stk[top][0]/stk[top][1] >= cur[0]/cur[1]){
            auto &j = stk[top];
            cur[0] += j[0], cur[1] += j[1];
            top -= 1;
        }
        stk[++ top] = cur;
    }

    auto &a = stk[top], &b = stk[0];
    cout << (ceil(a[0],a[1]) - b[0]/b[1]) << "\n";
}

int main(){

#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
