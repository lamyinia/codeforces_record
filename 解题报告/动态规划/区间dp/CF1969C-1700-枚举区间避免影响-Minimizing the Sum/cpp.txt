#include <bits/stdc++.h>
using namespace std;
const int mod = 1e9+7, inf = 1e9;

void solve(){
    int n, k;
    cin >> n >> k;
    vector <vector<long long>> f(n+1, vector<long long>(k+1));
    vector <int> A(n+1);
    for (int i = 1; i <= n; ++ i) cin >> A[i];
    // 需要 枚举至多操作次数， 再去枚举操作
    // 这里 相当于 是 枚举区间
    for (int i = 1; i <= n; ++ i){
        for (int j = 0; j <= k; ++ j){
            long long cur = f[i-1][j] + A[i];
            int Min = A[i];
            for (int l = i-1; l >= max(i-j, 1); -- l){
                Min = min(Min, A[l]);
                int t = i-l+1;
                cur = min(cur, f[l-1][j-t+1] + 1ll*Min*t);
            }
            f[i][j] = cur;
        }
    }

    cout << f[n][k] << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_ --)
        solve();
    return 0;
}
