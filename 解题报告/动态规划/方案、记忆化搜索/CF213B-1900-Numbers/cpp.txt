#include <bits/stdc++.h>
using namespace std;
const int mod = 1e9+7;
int C[101][101];

void solve(){
    for (int i = 0; i <= 100; ++ i){
        C[i][0] = 1;
        for (int j = 1; j <= i; ++ j){
            C[i][j] = (C[i-1][j] + C[i-1][j-1]) % mod;
        }
    }
    int n;
    array<int,10> req;
    cin >> n;
    for (int i = 0; i < 10; ++ i) cin >> req[i];
    vector <vector<int>> f(n+1, vector<int>(10, -1));
    auto dfs = [&](auto &&self, int i, int j)->int{
        if (f[i][j] >= 0) return f[i][j];
        if (j == 9){
            return i >= req[9];
        }
        int res = 0;
        for (int k = req[j]; k <= i; ++ k){
            res += 1ll*self(self, i-k, j+1) * C[i-(j==0)][k] % mod;
            res %= mod;
        }
        return f[i][j] = res;
    };

    int res = 0;
    for (int i = 1; i <= n; ++ i){
        res += dfs(dfs, i, 0);
        res %= mod;
    }
    cout << res << "\n";
}
 
int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
