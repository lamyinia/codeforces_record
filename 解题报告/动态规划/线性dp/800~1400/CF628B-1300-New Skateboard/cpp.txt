#include <bits/stdc++.h>
using ll = long long;
using namespace std;

void solve(){
    string s;
    cin >> s;
    int n = s.size();
    vector <vector<long long>> f(n+1, vector<long long>(4));

    long long res = 0;
    for (int i = 0; i < n; ++ i){
        int x = s[i]-'0';
        f[i+1][x%4] += 1;
        for (int j = 0, y; j < 4; ++ j){
            y = (j*10 + x) % 4;
            f[i+1][y] += f[i][j];
        }
        res += f[i+1][0];
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
