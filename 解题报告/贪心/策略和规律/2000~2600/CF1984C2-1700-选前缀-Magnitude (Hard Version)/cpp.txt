#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    const int mod = 998244353;

    int n;
    cin >> n;
    vector <int> A(n), pow2(n+1, 1);
    for (int i = 1; i <= n; ++ i) pow2[i] = pow2[i-1] * 2 % mod;
    for (auto &x : A) cin >> x;

    long long Min = 1e18, cur = 0;
    for (auto &x : A){
        cur += x;
        fmin(Min, cur);
    }

    if (Min >= 0){
        cout << pow2[n] << "\n";
        return;
    }

    cur = 0;
    long long res = 0;
    for (int i = 0, c = 0; i < n; ++ i){
        cur += A[i];
        if (cur >= 0) c += 1;
        if (cur == Min){
            res += pow2[c + n-i-1];
            if (res >= mod) res -= mod;
        }
    }
    
    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
