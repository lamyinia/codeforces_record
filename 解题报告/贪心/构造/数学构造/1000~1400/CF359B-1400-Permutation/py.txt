import sys

def main():
    n, k = map(int, input().split())
    A = [i for i in range(2*n, 0, -1)]
    for i in range(k):
        A[2*i], A[2*i+1] = A[2*i+1], A[2*i]
    print(*A)

if __name__ == '__main__':
    input = lambda: sys.stdin.readline().strip()
    _ = 1
    while _ != 0:
        _ -= 1
        main()
