#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector <int> A(n);
    for (auto &x : A) cin >> x;
    vector <vector<int>> f(n, vector<int>(20));

    for (int j = 0; 1<<j <= n; ++ j){
        for (int i = 0; i+(1<<j)-1 < n; ++ i){
            if (!j) f[i][j] = A[i];
            else f[i][j] = f[i][j-1] | f[i+(1<<j-1)][j-1];
        }
    }
    auto ask = [&](int l, int r)->int{
        int k = log2(r-l+1);
        return f[l][k] | f[r-(1<<k)+1][k];
    };

    auto check = [&](int mid)->bool{
        int cur = ask(0, mid-1);
        for (int i = 0; i < n-mid+1; ++ i){
            if (cur != ask(i, i+mid-1)){
                return false;
            }
        }
        return true;
    };

    int l = 1, r = n;
    while (l < r){
        int mid = (l+r) >> 1;
        if (check(mid)) r = mid;
        else l = mid+1;
    }

    cout << l << "\n";
}

int main(){

#else

#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
