#include <bits/stdc++.h>
using namespace std;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    int Max = -1e9;
    vector <int> A(n);
    for (auto &x : A){
        cin >> x;
        fmax(Max, x);
    }
    if (Max <= 0){
        cout << Max << "\n";
        return;
    }

    long long ans1 = 0, ans2 = 0;
    for (int i = 0; i < n; ++ i){
        if (A[i] <= 0) continue;
        if (i%2){
            ans2 += A[i];
        } else {
            ans1 += A[i];
        }
    }

    cout << max(ans1, ans2) << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
