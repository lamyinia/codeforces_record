#include <bits/stdc++.h>
using namespace std;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    using ull = unsigned long long;

    auto count = [&](long long x, long long dig)->long long{
        if (!dig) return 1;

        int len = 0;
        long long y = x;
        while (y){
            ++ len;
            y /= 10;
        }
        long long res = 0;
        if (dig <= x) res += 1;
        if (dig*10 + dig <= x) res += 1;

        if (len <= 2) return res;

        long long base = 10;
        for (int i = 3; i < len; ++ i){
            res += base;
            base *= 10;
        }
        
        if (dig*10*base + dig > x) return res;
        
        string s = to_string(x);
        
        if (int(s[0]-'0') > dig){
            res += base;
            return res;
        }

        for (int i = 1; i < len-1; ++ i){
            int c = s[i]-'0';
            base /= 10;
            if (c > 0){
                res += c * base;
            }
        }

        if (int(s[len-1]-'0') >= dig) res += 1;  // 考虑最后一位
        return res;
    };
    auto call = [&](long long x)->long long{
        long long res = 0;
        for (long long i = 0; i < 10; ++ i){
            res += count(x, i);
        }
        return res;
    };

    long long l, r;
    cin >> l >> r;

    cout << call(r)-call(l-1) << "\n";
}

int main(){

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
