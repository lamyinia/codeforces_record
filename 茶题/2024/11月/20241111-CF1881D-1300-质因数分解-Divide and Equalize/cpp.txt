const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector <int> A(n);
    map <int,int> mps;
    for (auto &x : A){
        cin >> x;
        for (int i = 2; 1ll*i*i <= x; ++ i){
            if (x%i==0){
                int cnt = 0;
                while (x%i==0){
                    x /= i;
                    cnt += 1;
                }
                mps[i] += cnt;
            }
        }
        if (x > 1) mps[x] += 1;
    }

    for (auto &it : mps){
        if (it.second%n){
            cout << "NO\n";
            return;
        }
    }
    cout << "YES\n";
}
