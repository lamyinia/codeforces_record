const int N = 1505, mod = 1e9 + 7, inf = 1e9;

void solve(){
    int n;
    cin >> n;
    string s, t;
    cin >> s >> t;
    s = " " + s, t = " " + t;
    int res = 0;
    for (int i = 1; i < n; ++ i){
        if (s[i] != t[i] && s[i+1] != t[i+1] && s[i]==t[i+1] && s[i+1]==t[i]){
            swap(s[i], s[i+1]);
            ++ res;
        }
    }
    for (int i = 1; i <= n; ++ i){
        if (s[i] != t[i]) ++ res;
    }

    cout << res << "\n";
}
