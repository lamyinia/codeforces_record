void fmax(int &x, int y){
    if (y > x) x = y;
}
void fmin(int &x, int y){
    if (y < x) x = y;
}

void solve(){
    int n;
    cin >> n;
    bool jg = 0;
    vector <int> A(n), L, R;
    for (auto &x : A){
        cin >> x;
        if (x) jg = 1;
        if (x <= 0) L.push_back(x);
        else R.push_back(x);
    }

    if (!jg){
        cout << "No\n";
        return;
    }

    vector <int> ans;

    sort(L.begin(), L.end(), greater<int>());
    sort(R.begin(), R.end());
    int up = R.back() - L.back();

    long long cur = 0;
    while (L.size() || R.size()){

        while (L.size() && cur >= 0){
            cur += L.back();
            ans.push_back(L.back());
            L.pop_back();
        }
        while (R.size() && cur < 0){
            cur += R.back();
            ans.push_back(R.back());
            R.pop_back();
        }

    }

    int len = ans.size();
    cout << "Yes\n";
    for (int i = 0; i < len; ++ i){
        if (i) cout << " ";
        cout << ans[i];
    }
    cout << "\n";
}
