#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector <long long> A(n+1), B(n+1), Pr(n+1);
    for (int i = 1; i <= n; ++ i) cin >> A[i];
    for (int i = 1; i <= n; ++ i) cin >> B[i], Pr[i] = Pr[i-1] + B[i];

    vector <long long> ds(n+1), ans(n+1);

    auto find = [&](int i)->int{
        int l = i, r = n;
        while (l < r){
            int mid = (l+r) >> 1;
            if (Pr[mid]-Pr[i-1] > A[i]) r = mid;
            else l = mid+1;
        }
        return l;
    };

    for (int i = 1; i <= n; ++ i){
        int j = find(i);
        ds[i] += 1;
        if (Pr[j]-Pr[i-1] > A[i]){
            ds[j] -= 1;
            ans[j] += A[i] - (Pr[j-1]-Pr[i-1]);
        }
    }
    for (int i = 1; i <= n; ++ i) ds[i] += ds[i-1];

    for (int i = 1; i <= n; ++ i){
        ans[i] += B[i] * ds[i];
    }

    for (int i = 1; i <= n; ++ i){
        if (i > 1) cout << " ";
        cout << ans[i];
    }
    cout << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
