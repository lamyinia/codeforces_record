#include <bits/stdc++.h>
using namespace std;
using ll = long long;
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}

void solve(){
    int n, k;
    cin >> n >> k;
    vector <long long> nums(n);
    for (auto &x : nums) cin >> x;

    if (k > 2) return cout << 0 << "\n", void();

    sort(nums.begin(), nums.end());
    auto call = [&](long long x)->long long{
        int idx = lower_bound(nums.begin(), nums.end(), x) - nums.begin();
        long long ans = 1e18;
        if (idx >= n) -- idx;
        fmin(ans, abs(x-nums[idx]));
        if (idx-1 >= 0) fmin(ans, abs(x-nums[idx-1]));
        return ans;
    };

    long long res = 1e18;
    for (int i = 0; i < n; ++ i){
        fmin(res, nums[i]);
        for (int j = i+1; j < n; ++ j){
            long long d = nums[j]-nums[i];
            fmin(res, d);
            if (k > 1) fmin(res, call(d));
        }
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
