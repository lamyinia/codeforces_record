#include <bits/stdc++.h>
using namespace std;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n, k;
    cin >> n >> k;
    vector<vector<int>> g(n);
    vector <int> fa(n), sz(n);
    for (int i = 0,u,v; i < n-1; ++ i){
        cin >> u >> v;
        u -= 1, v -= 1;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    vector <int> stk;
    stk.push_back(0);
    while (!stk.empty()){
        auto u = stk.back();
        stk.pop_back();
        for (auto &v : g[u]){
            if (v == fa[u]) continue;
            stk.push_back(v), fa[v] = u;
        }
    }
    
    auto dfs = [&](auto &&self, int u, int &rare, int mid)->void{
        sz[u] = 1;
        for (auto &v : g[u]){
            if (v == fa[u]) continue;
            self(self, v, rare, mid);
            if (rare > 0 && sz[v] >= mid) rare -= 1;
            else sz[u] += sz[v];
        }
    };
    auto chk = [&](int mid)->bool{
        int rare = k;
        dfs(dfs, 0, rare, mid);
        return rare == 0 && sz[0] >= mid;
    };


    int lo = 1, hi = n;
    while (lo < hi){
        int mid = (lo + hi + 1) >> 1;
        if (chk(mid)) lo = mid;
        else hi = mid-1;
    }

    cout << lo << "\n";
}

int main(){

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
