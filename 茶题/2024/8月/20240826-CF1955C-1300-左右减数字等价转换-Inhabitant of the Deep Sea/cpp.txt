#include <bits/stdc++.h>
using namespace std;
const int N = 3e5+11;
long long a[N], n, k;
long long L[N], R[N];

void solve(){
    cin >> n >> k;
    for (int i = 1; i <= n; ++ i)
        cin >> a[i], L[i] = a[i] + L[i-1];
    R[n+1] = 0;
    for (int i = n; i >= 1; -- i)
        R[i] = a[i] + R[i+1];
    if (L[n] <= k) return cout << n << "\n", void();
    long long lv = (k+1)/2, rv = k/2;
    int l = 1, r = n;
    int res = 0;
    while (L[l] <= lv) ++ l, ++ res;
    while (R[r] <= rv) -- r, ++ res;
    cout << res << "\n";
    return;
}

signed main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int T;
    cin >> T;
    while (T--) solve();
    return 0;
}
