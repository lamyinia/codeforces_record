#include <bits/stdc++.h>
using namespace std;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    long long n;
    cin >> n;

    long long tar = (n+1) / 2;
    auto chk = [&](long long mid)->bool{
        long long res = 0, cur = n;
        while (cur){
            res += min(cur, mid);
            cur -= min(cur, mid);
            cur -= cur / 10;
        }
        return res >= tar;
    };

    long long lo = 1, hi = n;
    while (lo < hi){
        long long mid = (lo+hi) >> 1;
        if (chk(mid)) hi = mid;
        else lo = mid+1;
    }

    cout << lo << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
