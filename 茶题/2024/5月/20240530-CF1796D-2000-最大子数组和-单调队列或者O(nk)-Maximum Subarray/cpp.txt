#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}

void solve(){
    const i64 inf = 1e18;

    int n, k, d;
    cin >> n >> k >> d;
    vector<vector<i64>> f(n+1, vector<i64>(k+1, -inf));
    
    vector <int> A(n);
    for (auto &x : A) cin >> x;

    f[0][0] = 0;
    for (int i = 0; i < n; ++ i){
        int x = A[i];
        for (int j = 0; j <= k; ++ j){
            if (f[i][j] == -inf) continue;

            fmax(f[i+1][j], max(1ll*x-d, f[i][j]+x-d));
            if (j < k){
                fmax(f[i+1][j+1], max(1ll*x+d, f[i][j]+x+d));
            }
        }
    }

    i64 res = 0;

    for (int i = 1; i <= n; ++ i){  // d 小于 0，选的越少越好
        int L = max(k - (n-i), 0);
        for (int j = L; j <= k; ++ j){
            fmax(res, f[i][j]);
        }
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(nullptr), cout.tie(nullptr);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
