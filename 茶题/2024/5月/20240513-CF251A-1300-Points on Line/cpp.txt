#include <bits/stdc++.h>
using namespace std;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n, d;
    cin >> n >> d;
    vector <int> A(n), got(n);
    for (auto &x : A) cin >> x;

    long long res = 0;
    for (int i = 0, j = -1; i < n-2; ++ i){
        while (j+1 < n && A[j+1]-A[i] <= d) j += 1;
        got[i] = j-i;
        if (got[i] > 1){
            res += 1ll * got[i] * (got[i]-1) / 2;
        }
    }
    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
