#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    auto sig = [&](int x)->int{
        return x > 0 ? 1 : -1;
    };

    i64 res = 0;
    int cur = 0, j1 = 0;
    for (int i = 0; i < n; ++ i){
        int x;
        cin >> x;
        int j2 = sig(x);
        if (j1 == 0){
            cur = x, j1 = j2;
        } else if (j1 == j2){
            fmax(cur, x);
        } else {
            res += cur;
            cur = x;
            j1 = j2;
        }
    }

    res += cur;
    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
