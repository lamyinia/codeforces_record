#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}

void solve(){
    const int mod = 998244353;

    int n;
    cin >> n;
    vector <i64> A(n), B(n);
    for (auto &x : A) cin >> x;
    for (auto &x : B) cin >> x;
    for (int i = 0; i < n; ++ i){
        A[i] = A[i] * (1ll * i * (n-i-1) + n);
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end(), greater<int>());

    i64 res = 0;
    for (int i = 0; i < n; ++ i){
        res += A[i]%mod * B[i] % mod;
        if (res >= mod) res -= mod;
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
