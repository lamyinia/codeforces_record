#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}


void solve(){
    string a, b;
    cin >> a >> b;
    int n = a.size();
    vector <int> Pr(n+1);
    for (int i = 0; i < n; ++ i){
        Pr[i+1] = Pr[i] + (a[i] == '1');
    }

    i64 res = 0;
    int m = b.size();
    for (int i = 0; i < m; ++ i){
        int r = min(i+1, n), l = max(i-(m-n), 0);
        int c1 = Pr[r]-Pr[l], c0 = (r-l) - c1;
        if (b[i] == '0') res += c1;
        else res += c0;
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(nullptr), cout.tie(nullptr);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
