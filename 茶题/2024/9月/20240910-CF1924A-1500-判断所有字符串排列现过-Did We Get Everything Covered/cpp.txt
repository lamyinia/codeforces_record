#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const long double Pi = acos(-1.0);
template <class T> void fmax(T &a, T b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, T b){
    if (b < a) a = b;
}

void solve(){
    int n, k, m;
    cin >> n >> k >> m;
    vector <int> cnt(k+10);

    string s;
    cin >> s;
    int c = 0;

    string res;
    for (int i = 0, all = 0; i < m; ++ i){
        if (!cnt[s[i]-'a']) all += 1;
        cnt[s[i]-'a'] += 1;

        if (all == k){
            res += s[i];
            fill(cnt.begin(), cnt.end(), 0);
            all = 0;
            c += 1;
        }

        if (c == n){
            cout << "YES\n";
            return;
        }
    }

    int j = 0;
    while (cnt[j]){
        j += 1;
    }
    assert(j < k);
    res += char(j+'a');
    while (res.size() < n) res += 'a';
    cout << "NO\n" << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
