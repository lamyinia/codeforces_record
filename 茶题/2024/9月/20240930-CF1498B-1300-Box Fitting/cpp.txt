#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int limi = 1e9;
map <int, int> mp2;

void solve(){
    int n, all;
    cin >> n >> all;
    vector <int> cnt(20);
    for (int i = 0, x; i < n; ++ i){
        cin >> x;
        cnt[mp2[x]] += 1;
    }

    int siz = n;
    int res = 0;
    while (siz){
        int cur = all;
        for (int i = 19; i >= 0; -- i){
            while (cnt[i] && cur - (1<<i) >= 0){
                cnt[i] -= 1, siz -= 1;
                cur -= 1<<i;
            }
        }
        res += 1;
    }

    cout << res << "\n";
}
 
int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    for (int i = 0; i < 20; ++ i) mp2[1<<i] = i;
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
