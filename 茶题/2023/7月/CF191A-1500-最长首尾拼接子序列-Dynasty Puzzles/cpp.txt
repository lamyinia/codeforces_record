#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
using f64 = long double;
const f64 eps = 1e-9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}

void solve(){
    using ary3 = array<int,3>;

    int n;
    cin >> n;
    vector <ary3> gs;
    for (int i = 0; i < n; ++ i){
        string s;
        cin >> s;
        int siz = s.size();
        gs.push_back({s[0]-'a', s[siz-1]-'a', siz});
    }

    vector<vector<int>> f(26, vector<int>(26, -1));
    for (int i = 0; i < n; ++ i){
        int u1 = gs[i][0], u2 = gs[i][1], val = gs[i][2];
        vector <int> tmps;
        for (int j = 0; j < 26; ++ j){
            tmps.push_back(f[j][u1]);
        }
        for (int j = 0; j < 26; ++ j){
            if (tmps[j] != -1){
                fmax(f[j][u2], tmps[j]+val);
            }
        }

        fmax(f[u1][u2], val);
    }

    int res = 0;
    for (int i = 0; i < 26; ++ i){
        fmax(res, f[i][i]);
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
