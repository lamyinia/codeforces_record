#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}

void solve(){
    int n;
    cin >> n;
    vector <int> A(n);

    int cnt[3] = {};
    for (auto &x : A){
        char ch;
        cin >> ch;
        x = ch - '0';
        cnt[x] += 1;
    }

    int tar = n/3;

    if (cnt[0] < tar){
        for (int i = 0; i < n && cnt[0] < tar; ++ i){
            if (A[i] != 0 && cnt[A[i]] > tar){
                cnt[A[i]] -= 1;
                A[i] = 0;
                cnt[0] += 1;
            }
        }
    }
    if (cnt[2] < tar){
        for (int i = n-1; i >= 0 && cnt[2] < tar; -- i){
            if (A[i] != 2 && cnt[A[i]] > tar){
                cnt[A[i]] -= 1;
                A[i] = 2;
                cnt[2] += 1;
            }
        }
    }
    if (cnt[1] < tar){
        for (int i = n-1; i >= 0 && cnt[1] < tar; -- i){
            if (A[i] == 0 && cnt[A[i]] > tar){
                cnt[A[i]] -= 1;
                A[i] = 1;
                cnt[1] += 1;
            }
        }
        for (int i = 0; i < n && cnt[1] < tar; ++ i){
            if (A[i] == 2 && cnt[A[i]] > tar){
                cnt[A[i]] -= 1;
                A[i] = 1;
                cnt[1] += 1;
            }
        }
    }

    for (auto &x : A) cout << x;
    cout << "\n";
    return;
}

int main(){

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
