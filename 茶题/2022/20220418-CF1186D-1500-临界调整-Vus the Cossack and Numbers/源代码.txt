#include <bits/stdc++.h>
using namespace std;
using Pii = pair<int,int>;
using ll = long long;
const int N = 1e5+5, inf = (1u<<31)-1, mod = 998244353;
const double eps = 1e-9;

void solve(){
    int n;
    cin >> n;
    vector <double> nums(n+1);
    vector <int> ans(n+1);
    for (int i = 1; i <= n; ++ i) cin >> nums[i];
    ll tot = 0;
    for (int i = 1; i <= n; ++ i){
        ans[i] = floor(nums[i]);
        tot += ans[i];
    }
    if (tot < 0)
        for (int i = 1; i <= n; ++ i){
            if (fabs(nums[i] - floor(nums[i])) > eps){
                ++ ans[i];
                ++ tot;
            }
            if (tot == 0) break;
        }
    for (int i = 1; i <= n; ++ i) cout << ans[i] << "\n";
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int _ = 1;
    while (_--) solve();
    return 0;
}
