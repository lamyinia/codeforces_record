#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int INF = 2e9;
const int N = 100050;

void solve(){
    ll n, k, x, op;
    cin >> n >> k >> x >> op;
    vector <ll> speed(n), t(k+1), p(k+1), dic;
    // 小于等于
    for (int i = 1; i <= n; ++ i) cin >> speed[i-1], dic.push_back(speed[i-1]);
    for (int i = 1; i <= k; ++ i) cin >> t[i];
    for (int i = 1; i <= k; ++ i) cin >> p[i];
    sort(speed.begin(), speed.end());
    sort(dic.begin(), dic.end());
    t[0] = 0, p[0] = op;

    int dis = x, res = 0;
    for (int i = 0, snap; i <= k; ++ i){
        snap = p[i]-t[i];
        int need = int(ceil(1.0*dis / snap));
        while (speed.size() && speed.back() >= need) ++ res, speed.pop_back();
    }
    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
