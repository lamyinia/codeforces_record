#include <bits/stdc++.h>
using namespace std;
using ll = long long;

void solve(){
    vector <int> b(3), d(3);
    for (int i = 0; i < 3; ++ i) cin >> b[i];
    for (int i = 0; i < 3; ++ i) cin >> d[i];

    ll res = 0;
    for (int i = 0; i < 3; ++ i){
        if (i == 0){
            if (d[1] >= b[0]) d[1] -= b[0], res += b[0];
            else {
                int delta = b[0] - d[1];
                res += d[1], d[1] = 0;
                if (d[0] >= delta) d[0] -= delta;
                else res += d[0]-delta, d[2] += d[0]-delta, d[0] = 0;
            }
        }

        if (i == 1){
            if (d[2] >= b[1]) d[2] -= b[1], res += b[1];
            else {
                int delta = b[1] - d[2];
                res += d[2], d[2] = 0;
                if (d[1] >= delta) d[1] -= delta;
                else res += d[1]-delta, d[0] += d[1]-delta, d[1] = 0;
            }
        }

        if (i == 2){
            if (d[0] >= b[2]) d[0] -= b[2], res += b[2];
            else {
                int delta = b[2] - d[0];
                res += d[0], d[0] = 0;
                if (d[2] >= delta) d[2] -= delta;
                else res += d[2]-delta, d[1] += d[2]-delta, d[2] = 0;
            }
        }
    }
    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
