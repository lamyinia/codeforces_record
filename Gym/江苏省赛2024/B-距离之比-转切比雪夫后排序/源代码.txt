#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int N = 1e5+11, inf = 0x3f3f3f3f;

void solve(){
    int n;
    cin >> n;
    vector <ll> x(n+1), y(n+1), p(n+1);
    for (int i = 1; i <= n; ++ i) cin >> x[i] >> y[i];
    iota(p.begin(), p.end(), 0);
    double res = 1.0;  //最小值为1
    for (int t = 0; t < 2; ++ t){
        sort(p.begin()+1, p.end(), [&](int i, int j){
            return x[i] + (t?-1:1)*y[i] < x[j] + (t?-1:1)*y[j];
        });
        for (int i = 2; i <= n; ++ i){
            int a = p[i], b = p[i-1];
            ll dis1 = abs(x[a]-x[b]) + abs(y[a]-y[b]);
            double dis2 = hypot(x[a]-x[b], y[a]-y[b]);
            res = max(res, dis1/dis2);
        }
    }
    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    cout << fixed << setprecision(9);
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
