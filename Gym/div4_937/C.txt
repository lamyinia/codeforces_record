#include <bits/stdc++.h>
using namespace std;

void solve(){
    string str;
    cin >> str;
    int x = (str[0]-'0')*10 + str[1]-'0';
    int sign = 0;
    if (x >= 12) x -= 12, sign = 1;
    if (!x) x += 12;
    string res;
    if (x < 10) res += "0"+to_string(x)+str.substr(2);
    else res += to_string(x)+str.substr(2);
    
    if (sign) cout << res << " PM\n";
    else cout << res << " AM\n"; 
}

int main(){
    int T;
    cin >> T;
    while (T--) solve();
    return 0;
}
