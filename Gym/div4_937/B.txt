#include <bits/stdc++.h>
using namespace std;

void solve(){
    int n;
    cin >> n;
    for (int i = 1, isign=1; i <= 2*n; ++ i){
        
        for (int j = 1, jsign=1; j <= 2*n; ++ j){
            if (isign*jsign > 0)  cout << "#";
            else cout << ".";
            if (j%2==0) jsign = -jsign;
        }
        cout << "\n";
        if (i%2==0) isign = -isign;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T;
    cin >> T;
    while (T--) solve();
    return 0;
}
