#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int N = 1e5+11, inf = 0x3f3f3f3f;

void solve(){
    ll l, r;
    cin >> l >> r;
    ll res = 0;
    for (ll i = 1; i*i <= r; ++ i){
        ll t = r/i;
        if ((l-1)/i + 1 < r/i) res = max(res, i);
        if ((l-1)/t + 1 < r/t) res = max(res, t);
    }
    cout << res << "\n";
}
// 联想到x和y一定是gcd*k和gcd*（k+1），然后gcd和k不会同时超过1e6，分别枚举取最优就好了
