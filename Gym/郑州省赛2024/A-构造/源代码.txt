#include <bits/stdc++.h>
#define int long long
#define endl '\n'
using ll = long long;
typedef unsigned long long ull;
using namespace std;
 
void GordenGhost() {
    ll n,d,ans=123456789;
    cin>>n>>d;
    ans=ans*10+d;
    string s=to_string(n);
    ans=ans* pow(10,s.size());
    ans+=n-ans%n;
    cout<<ans/n<<endl;
}
 
signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);
    int t;
    cin>>t;
    while (t--)
        GordenGhost();
    return 0;
}
