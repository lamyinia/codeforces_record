#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int mod = 1e9+7;

void solve(){
    int n;
    cin >> n;
    vector <int> dg(n);
    for (int i = 0, a, b; i < n-1; ++ i){
        cin >> a >> b;
        -- a, -- b;
        ++ dg[a], ++ dg[b];
    }
    string s;
    cin >> s;
    bool f = s[0] != '?';
    int c0 = 0, c1 = 0, cp = 0, cq = 0;
    for (int i = 1; i < n; ++ i){
        if (dg[i] == 1){
            if (s[i] == '0') ++ c0;
            if (s[i] == '1') ++ c1;
            if (s[i] == '?') ++ cp;
        } else if (s[i] == '?') ++ cq;
    }

    if (f){
        if (s[0] == '1') cout << c0 + (cp+1)/2 << "\n";
        else cout << c1 + (cp+1)/2 << "\n";
    } else {
        if (c0 != c1) cout << max(c0, c1) + cp/2 << "\n";
        else if (cp%2) cout << c0 + (cp + cq%2)/2 << "\n";
        else cout << c0 + cp/2 << "\n";
    }
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_ --)
        solve();
    return 0;
}
