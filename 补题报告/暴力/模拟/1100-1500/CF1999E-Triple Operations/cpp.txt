const int N = 2e5, mod = 1e9+7, inf = 1e9;

void solve(){
    int l, r;
    cin >> l >> r;

    int ans = 0, low = -1;
    for (int sum = 1, t = 1; sum <= r; sum *= 3, ++ t){
        if (l >= sum){
            low = t;
        }
        ans += max(0, (min(3*sum-1, r) - max(sum, l) + 1) * t);

    }
    ans += low;

    cout << ans << "\n";
}

        solve();

}
