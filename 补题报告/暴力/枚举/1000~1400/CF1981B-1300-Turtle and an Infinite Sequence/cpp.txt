#include <bits/stdc++.h>
using ll = long long;
using namespace std;

void solve(){

    auto find = [&](long long x)->int{
        int res = 0;
        for (int i = 0; i <= 31; ++ i){
            if ((x>>i)&1){
                res = i;
            }
        }
        return res;
    };

    long long n, m;
    cin >> n >> m;
    int idx = find(n);  // 找最高位1的位置
    long long res = n;
    for (int i = (n == 0 ? 0 : idx+1); i <= 31; ++ i){
        if ((1ll<<i)-n <= m){
            res |= 1ll<<i;
        }
    }

    long long cur = 1ll << idx;
    for (int i = idx-1, lst = idx; i >= 0; -- i){
        if ((n>>i)&1){
            lst = i, cur |= 1ll << i;
        } else {
            long long nxt = (1ll<<lst)-1;
            long long x = (cur|(1ll<<i)) - n, y = n - ((n-(1ll<<lst))|nxt);
            if (min(x, y) <= m){
                res |= 1ll<<i;
            }
        }
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
