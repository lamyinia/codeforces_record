#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int N = 5050, p = 998244353, INF = 2e9;
int n;
vector <ll> a(N), b(N);

void solve(){
    cin >> n;
    for (int i = 1; i <= n; ++ i) cin >> a[i];

    ll res = 9e18;
    for (int i = 1; i <= n; ++ i){
        ll cnt = 0;
        fill(b.begin()+1, b.begin()+1+n, 0);
        for (int j = i-1; j >= 1; -- j){
            ll k = ceil(1.0*b[j+1]/a[j]) - 1;  //小于上取整,建议用ceil不会爆ll，找到负数的最大值
            b[j] = k*a[j];
            cnt += -k;
        }
        for (int j = i+1; j <= n; ++ j){
            ll k = b[j-1]/a[j] + 1;  //大于下取整，找到正数的最小值
            b[j] = k*a[j];
            cnt += k;
        }
        res = min(res, cnt);
    }

    cout << res << "\n";
}
 
int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
