#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int N = 200050;

void solve(){
    ll n, a, b;
    cin >> n >> a >> b;
    vector <ll> nums(n+1);
    vector <ll> Sum(n+1);
    for (int i = 1; i <= n; ++ i) cin >> nums[i];
    for (int i = 1; i <= n; ++ i) Sum[i] = Sum[i-1] + nums[i];

    auto Func = [&](int pos)->ll{
        return (a+b)*nums[pos] + b*(Sum[n]-Sum[pos] - (n-pos)*nums[pos]);
    };
    ll res = 9e18;
    for (int i = 0; i <= n; ++ i)
        res = min(res, Func(i));

    cout << res << "\n";
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
