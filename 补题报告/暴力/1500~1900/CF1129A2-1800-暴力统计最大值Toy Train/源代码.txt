/*
环上有n个点，火车在环上运输糖果。
给定m个糖果，需要从第 ai 个点运到第 bi个点。
火车容量无限大，但是在每个点只能装上一个糖果。
分别输出火车以 第 i 个点为起点完成运输任务的最小时间。
*/
#include <bits/stdc++.h>
using namespace std;

void solve(){
	int n, m;
	cin >> n >> m;
	vector <int> dis(n+1, 2e9), cnt(n+1);
	
	for (int i = 1; i <= m; ++ i){
		int a, b;
		cin >> a >> b;
		int d = b < a ? n+b-a : b-a;
		dis[a] = min(dis[a], d), ++ cnt[a];
	}
	
	for (int i = 1; i <= n; ++ i){
		long long res = 0;
		for (int j = 1; j <= n; ++j)
			if (cnt[j]) res = max(res, 1ll*(cnt[j]-1)*n + (j-i+n)%n + dis[j]);  //贪心取最大值，答案只和该点的运输最小距离和运输次数有关
		cout << res << " ";
	}
	cout << "\n";
}

int main(){
	int _ = 1;
	while (_--) solve();
	return 0;
}
