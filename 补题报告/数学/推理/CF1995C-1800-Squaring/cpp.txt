#include <bits/stdc++.h>
using namespace std;
const double eps = 1e-12;

void solve(){
    long long n, lst;
    cin >> n >> lst;
    vector <long long> nums(n-1);
    for (auto &x : nums) cin >> x;
    long long res = 0, cur = 0;
    for (auto &x : nums){
        if (x == 1){
            if (lst == 1) continue;
            res = -1;
            break;
        }
        cur += max(-1.0*cur, ceil(log2(log2(lst) / log2(x)) - eps));
        res += cur;
        lst = x;
    }

    cout << res << "\n";
}
 
int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif

    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
