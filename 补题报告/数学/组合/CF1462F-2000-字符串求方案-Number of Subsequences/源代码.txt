#include <bits/stdc++.h>
using namespace std;
using Pii = pair<int,int>;
using ll = long long;
const int N = 1e5+5, inf = (1u<<31)-1, mod = 1e9+7;
const double eps = 1e-9;

/*
    对'a','c','?'做前缀和
*/
void solve(){
    int n;
    cin >> n;
    string s;
    cin >> s;
    s = " " + s;
    vector <array<int,3>> f(n+1);
    vector <int> pow3(n+1,1);
    for (int i = 1; i <= n; ++ i){
        f[i][0] = f[i-1][0] + (s[i]=='a');
        f[i][1] = f[i-1][1] + (s[i]=='c');
        f[i][2] = f[i-1][2] + (s[i]=='?');
        pow3[i] = 3ll*pow3[i-1]%mod;
    }

    int res = 0;
    for (int i = 1; i <= n; ++ i){
        if (s[i]=='b' || s[i]=='?'){
            int ct1 = f[i-1][0], ct2 = f[n][1] - f[i][1]; // a, c
            int ct3 = f[i-1][2], ct4 = f[n][2]-f[i][2];  // l? r?
            res += 1ll*ct1*ct2*pow3[ct3+ct4]%mod, res %= mod;
            res += 1ll*ct1*ct4*pow3[ct3+ct4-1]%mod, res %= mod;
            res += 1ll*ct3*ct2*pow3[ct3+ct4-1]%mod, res %= mod;
            res += 1ll*ct3*ct4*pow3[ct3+ct4-2]%mod, res %= mod;
        }
    }

    cout << res << "\n";
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int _ = 1;
    while (_--) solve();
    return 0;
}
