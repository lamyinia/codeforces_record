#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using Pii = pair<int,int>;
const int N = 400550, mod = 998244353, inf = 1e9;
int nxt[N];
struct cmp {
    bool operator()(array<int,2> &a, array<int,2> &b){
        return a[0] < b[0];
    }
};
bool vis[N];
void solve(){
    int n, k;
    cin >> n >> k;
    vector <int> a(n+1);
    for (int i = 1; i <= n; ++ i) cin >> a[i];
    vector <int> Pos(n+1);
    for (int i = 1; i <= n; ++ i) Pos[i] = n+1;
    for (int i = n; i >= 1; -- i){
        nxt[i] = Pos[a[i]];
        Pos[a[i]] = i;
    }
    int res = 0;
    set <Pii> st;
    for (int i = 1, cnt = 0; i <= n; ++ i){
        int x = a[i];
        if (vis[x]) st.erase({i, x}), st.insert({nxt[i], x});
        else {
            ++ res;
            if (st.size() == k){
                auto p = st.rbegin();
                vis[p->second] = false;
                st.erase(*p);
            }
            st.insert({nxt[i], x});
            vis[x] = true;
        }
    }
    cout << res << "\n";
}

int main(){

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_--) solve();
    return 0;
}
