    int pos = 0;
    for (int i = 1; i <= k+1; ++ i){
        cout << "? ";
        for (int j = 1; j <= k+1; ++ j)
            if (j != i) cout << j << " ";
        cout.flush();

        cin >> x >> y;
        ++ mp[y];

}
