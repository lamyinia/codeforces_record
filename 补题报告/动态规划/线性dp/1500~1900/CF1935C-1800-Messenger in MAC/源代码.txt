#include <bits/stdc++.h>
using namespace std;
using Pii = pair<int,int>;
using ll = long long;
const int N = 4e5+10, mod = 1e9 + 7, inf = 1e9;
/*
    f[i][j] 表示前i个数选j个, 且第j个必选的最小价值
*/
void solve(){
    ll n, l;
    cin >> n >> l;
    vector<vector<ll>> f(n+1, vector<ll>(n+1, 1e18));
    vector <array<int,2>> nums(n+1);
    for (int i = 1; i <= n; ++ i) cin >> nums[i][0] >> nums[i][1];
    sort(nums.begin()+1, nums.end(), [&](array<int,2> &a, array<int,2> &b){
        return a[1] < b[1];
    });
    vector<ll> last(n+1, 1e18);
    last[0] = 0;
    int res = 0;
    for (int i = 1; i <= n; ++ i){
        for (int j = 1; j <= i; ++ j){
            if (j==1) f[i][j] = nums[i][0];
            else {
                f[i][j] = last[j-1] + nums[i][0] + nums[i][1];
            }
            if (f[i][j] <= l) res = max(res, j);
        }
        for (int j = 0; j <= i; ++ j)    
            last[j] = min(last[j], f[i][j]-nums[i][1]);
    }
    cout << res << "\n";
}

int main(){

#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--)
        solve();
    return 0;
}

#include <bits/stdc++.h>

    f[i][j] 表示前i个数选j个, 且第j个必选的最小价值
*/
void solve(){
    ll n, l;
    cin >> n >> l;
    vector<ll> f(n+1);
    vector <array<int,2>> nums(n+1);
    for (int i = 1; i <= n; ++ i) cin >> nums[i][0] >> nums[i][1];
    sort(nums.begin()+1, nums.end(), [&](array<int,2> &a, array<int,2> &b){
        return a[1] < b[1];
    });
    vector<ll> last(n+1, 1e18);
    last[0] = 0;
    int res = 0;
    for (int i = 1; i <= n; ++ i){
        for (int j = i; j >= 1; -- j){
            if (j==1) f[j] = nums[i][0];
            else {
                f[j] = last[j-1] + nums[i][0] + nums[i][1];
            }
            last[j] = min(last[j], f[j]-nums[i][1]);
            if (f[j] <= l) res = max(res, j);
        }
    }
    cout << res << "\n";
}
