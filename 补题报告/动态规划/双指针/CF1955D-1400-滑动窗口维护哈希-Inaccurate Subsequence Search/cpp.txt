
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

void solve(){
    int n, m, k;
    cin >> n >> m >> k;
    vector <int> A(n);
    map <int,int> mpR, mpL;
    for (auto &x : A) cin >> x;
    for (int i = 0, x; i < m; ++ i){
        cin >> x;
        mpR[x] += 1;
    }

    int cur = 0;
    auto add = [&](int x){
        if (mpR.count(x)){
            if (mpL[x] < mpR[x]) cur += 1;
            mpL[x] += 1;
        }
    };
    auto dele = [&](int x){
        if (mpR.count(x)){
            mpL[x] -= 1;
            if (mpL[x] < mpR[x]) cur -= 1;
        }
    };

    int res = 0;
    for (int i = 0, j = -1; i < n; ++ i){
        while (j+1 < n && j-i+1 < m){
            j += 1;
            add(A[j]);
        }
        if (j-i+1 != m) break;
        if (cur >= k) res += 1;
        dele(A[i]);
    }

    cout << res << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
