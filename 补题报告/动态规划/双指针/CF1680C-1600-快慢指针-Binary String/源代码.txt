#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int N = 1050, mod = 1e9+7, inf = 1e9;

void solve(){
    string s;
    cin >> s;
    int n = s.size();
    s = " " + s;
    int ct1 = 0, ct2 = 0;
    for (int i = 1; i <= n; ++ i) ct2 += s[i]=='1';

    int res = inf;
    //留下来0的个数，删掉的1的个数，固定左端点
    //指针j一定单调右移
    for (int i = 1, j = 0; i <= n; ++ i){
        while (ct1 < ct2 && j+1 <= n){
            if (s[j+1] == '0') ++ ct1;
            else -- ct2;
            ++ j;
        }
        res = min(res, max(ct1,ct2));
        if (s[i] == '0') -- ct1;
        else ++ ct2;
    }
    cout << res << "\n";
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
