#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int mod = 1e9+7;

void solve(){
    string s;
    cin >> s;
    int n = s.size();
    s = " " + s;
    vector <int> ne(n+1);
    ne[0] = -1;
    for (int i = 1, j; i <= n; ++ i){
        j = ne[i-1];
        while (j!=-1 && s[j+1] != s[i]) j = ne[j];
        ne[i] = j+1;
    }

    if (ne[n] == 0 || ne[ne[n]] <= 0 || n%ne[n] == 0){
        cout << "NO\n";
        return;
    }
    int det = n-ne[n];
    for (int i = min(det,ne[n])+1; i <= n; ++ i){
        if (s[i] != s[i-min(det,ne[n])]){
            cout << "NO\n";
            return;
        }
    }

    cout << "YES\n" << s.substr(1, ne[n]) << "\n";
}

int main(){
#ifdef ONLINE_JUDGE
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#else
    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);
#endif
    int _ = 1;
    while (_ --)
        solve();
    return 0;
}
