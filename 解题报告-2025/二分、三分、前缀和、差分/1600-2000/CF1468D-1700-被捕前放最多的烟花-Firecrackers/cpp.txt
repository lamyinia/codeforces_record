const f64 eps = 1e-12;

const int inf_int = 1e9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int n, m, a, b;
    cin >> n >> m >> a >> b;
    int d1 = abs(b-a) - 1;
    int d2 = a < b ? b-1 : n-b;

    vector <int> A(m);
    for (auto &x : A){
        cin >> x;
    }
    sort(A.begin(), A.end());

    auto chk = [&](int mid)->bool{
        for (int i = mid-1, cur = 1; i >= 0; -- i, cur += 1){
            if (A[i]+cur > d2){
                return false;
            }
        }
        return true;
    };

    int lo = 0, hi = min(m, d1);
    while (lo < hi){
        int mid = (lo+hi+1) >> 1;
        if (chk(mid)) lo = mid;
        else hi = mid-1;
    }

    cout << lo << "\n";
}
