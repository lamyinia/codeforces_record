#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
using f64 = long double;
using u64 = unsigned long long;
const f64 eps = 1e-12;
const i64 inf_i64 = 1e18;
const int inf_int = 1e9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    i64 n, k, all = 0;
    cin >> n >> k;
    vector <i64> A(n);
    for (auto &x : A){
        cin >> x;
        all += x;
    }

    if (k > all) return void(cout << -1 << "\n");

    auto chk = [&](i64 mid)->bool{
        i64 res = 0;
        for (auto &x : A){
            res += min(x, mid);
        }
        return res >= k;
    };
    i64 lo = 1, hi = k;
    while (lo < hi){
        i64 mid = (lo + hi) >> 1;
        if (chk(mid)) hi = mid;
        else lo = mid+1;
    }

    i64 cur = 0;
    for (int i = 0; i < n; ++ i){
        i64 x = min(A[i], lo-1);
        cur += x;
        A[i] -= x;
    }

    int mark = -1;
    if (k > 0){
        for (int i = 0; i < n; ++ i){
            if (A[i]){
                A[i] -= 1;
                cur += 1;
            }
            if (cur == k){
                mark = i;
                break;
            }
        }
    }

    // assert(cur == k);

    vector <int> ans;
    for (int i = mark+1; i < n; ++ i){
        if (A[i]){
            ans.push_back(i+1);
        }
    }
    for (int i = 0; i <= mark; ++ i){
        if (A[i]){
            ans.push_back(i+1);
        }
    }

    print(ans);
}

#endif
    int _ = 1;
    while (_--){
        solve();

}
