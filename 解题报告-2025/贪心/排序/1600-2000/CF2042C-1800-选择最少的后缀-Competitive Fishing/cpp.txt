using f64 = long double;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}

void solve(){
    int n, k;
    cin >> n >> k;
    vector <int> A(n);
    for (int i = 0; i < n; ++ i){
        char ch;
        cin >> ch;
        A[i] = ch == '1' ? 1 : -1;
    }

    priority_queue <int> pq;
    for (int i = n-1, cur = 0; i >= 1; -- i){
        cur += A[i];
        pq.push(cur);
    }

    int res = 0, cur = 0;
    while (!pq.empty()){
        int x = pq.top();
        pq.pop();
        if (x < 0) break;
        cur += x, res += 1;
        if (cur >= k) break;
    }
    // cout << cur << "\n";
    if (cur >= k) cout << res+1 << "\n";
    else cout << -1 << "\n";
}

#endif
    int _ = 1;
    cin >> _;
    while (_--) solve();
    return 0;
}
