#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
using f64 = long double;
using u64 = unsigned long long;
const f64 eps = 1e-12;
const i64 inf_i64 = 4e18;
const int inf_int = 1e9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int n;
    cin >> n;
    vector<vector<int>> chuks(n+1);
    int l = 0, r = n+1, ul = n+1, ur = 0;
    for (int i = 1,x; i <= n; ++ i){
        cin >> x;
        fmax(l, i-x+1), fmin(r, i+x-1);
        chuks[x].push_back(i);
    }
    bool jg = 1;
    for (int i = 1; i <= n && jg; ++ i){
        for (auto &x : chuks[i]){
            fmin(ul, x), fmax(ur, x);
        }
        if (ur-ul+1 > i){
            jg = 0;
        }
    }

    if (jg && l <= r) cout << r-l+1 << "\n";
    else cout << 0 << "\n";
}

    int _ = 1;
    cin >> _;
    while (_--){
        solve();
    }
    return 0;
}
