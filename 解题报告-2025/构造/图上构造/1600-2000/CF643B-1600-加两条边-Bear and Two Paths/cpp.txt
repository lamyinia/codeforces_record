#include <bits/stdc++.h>using namespace std;using i64 = long long;using f64 = long double;using u64 = unsigned long long;const f64 eps = 1e-12;const i64 inf_i64 = 4e18;const int inf_int = 1e9;template <class T> void fmax(T &a, const T &b){    if (b > a) a = b;}template <class T> void fmin(T &a, const T &b){    if (b < a) a = b;}template <class T> void print(vector <T> &A, int d = 0){    int n = A.size();    for (int i = 0; i < n; ++ i){        if (i) cout << " ";        cout << A[i]+d;    }    cout << "\n";}int jgs(f64 x){    return x < -eps ? -1 : x > eps;}void solve(){    int n, k, a, b, c, d;    cin >> n >> k;    cin >> a >> b >> c >> d;        if (k <= n || n == 4)        return void(cout << -1 << "\n");        vector <int> ans1(n), ans2 = {c, a};    ans1[0] = a, ans1[1] = c, ans1[n-2] = d, ans1[n-1] = b;    for (int i = 1, j = 1; i <= n; ++ i){        if (i == a || i == b || i == c || i == d) continue;        ans1[++ j] = i;       }    for (int i = 2; i < n-2; ++ i){        ans2.push_back(ans1[i]);    }    ans2.push_back(b);    ans2.push_back(d);        print(ans1);    print(ans2);}int main(){#ifdef ONLINE_JUDGE    ios::sync_with_stdio(0), cin.tie(nullptr), cout.tie(nullptr);#else    freopen("in.txt", "rt", stdin), freopen("out.txt", "wt", stdout);#endif    int _ = 1;    while (_--){        solve();    }    return 0;}
