    int n = A.size();

        if (i) cout << " ";

int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int n, m;
    cin >> n >> m;
    vector<queue<int>> Q(26);
    string s, t;
    cin >> s >> t;
    for (int i = 0; i < n; ++ i){
        Q[s[i]-'a'].push(i);
    }

    for (int i = 0; i < m; ++ i){
        int chose = t[i]-'a';
        if (Q[chose].empty()){
            cout << "NO\n";
            return;
        }
        int id = Q[chose].front();
        Q[chose].pop();
        for (int i = 0; i < chose; ++ i){
            while (!Q[i].empty() && Q[i].front() < id){
                Q[i].pop();
            }
        }
    }

    cout << "YES\n";
    return;
}
