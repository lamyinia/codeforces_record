const f64 eps = 1e-12;

template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int n, m, k;
    cin >> n >> m >> k;
    vector <int> A(n), vis(n);
    for (auto &x : A){
        cin >> x;
    }
    sort(A.begin(), A.end());

    for (int i = 0, j = -1, cur = 0; i < n; ++ i){
        if (!vis[i]){
            while (i > j || j+1 < n && A[j+1]-A[i]+1 <= m){
                j += 1;
                cur += 1;
                if (cur >= k){
                    vis[j] = 1;
                    cur -= 1;
                }
            }
        }
        cur -= !vis[i];
    }

    int res = 0;
    for (int i = 0; i < n; ++ i){
        res += vis[i];
    }

    cout << res << "\n";
}
