#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
using f64 = long double;
using u64 = unsigned long long;
const f64 eps = 1e-12;
const i64 inf_i64 = 4e18;
const int inf_int = 1e9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int n, h;
    cin >> h >> n;
    vector <int> A(n+2);
    for (int i = 1; i <= n; ++ i){
        cin >> A[i];
    }
    vector <int> f(n+2, inf_int);
    f[1] = 0;
    for (int i = 2; i <= n+1; ++ i){
        fmin(f[i], f[i-1] + (i <= n));
        if (i-2 >= 1 && A[i]+1 == A[i-1]){
            fmin(f[i], f[i-2]);
        }
    }
    cout << f[n+1] << "\n";
}
