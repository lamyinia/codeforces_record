#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
using f64 = long double;
using u64 = unsigned long long;
const f64 eps = 1e-12;
const i64 inf_i64 = 4e18;
const int inf_int = 1e9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int n;
    cin >> n;
    vector <int> A(n);
    for (auto &x : A){
        cin >> x;
    }
    vector<vector<i64>> f(n,vector<i64>(n));
    for (int r = 2; r < n; ++ r){
        for (int l = r-2; l >= 0; -- l){
            for (int i = l+1; i < r; ++ i){
                fmax(f[l][r], f[l+1][i-1]+f[i+1][r-1] + A[l]*A[i]*A[r]);
            }
            for (int i = 1; i < r; ++ i){
                fmax(f[l][r], f[l][i] + f[i+1][r]);
            }
        }
    }

    cout << f[0][n-1] << "\n";
}
