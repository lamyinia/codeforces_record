template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    const int mod = 1e9+7;
    auto pows = [&](i64 a, int b)->int{
        i64 res = 1;
        while (b){
            if (b&1) res = res*a % mod;
            a = a*a % mod;
            b >>= 1;
        }
        return res;
    };
    int n, m, k;
    cin >> n >> m >> k;

    int res = 0;
    if (k == 1 || k > n) res = pows(m, n);
    else if (k == n) res = pows(m, (n+1)/2);
    else if (k%2) res = m*m;
    else res = m;

    cout << res << "\n";
}
