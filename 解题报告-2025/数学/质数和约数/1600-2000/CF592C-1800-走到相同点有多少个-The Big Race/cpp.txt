    if (b > a) a = b;
}

    if (b < a) a = b;

template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    i64 L, w, b;
    cin >> L >> w >> b;
    if (w < b) swap(w, b);

    i64 l = (__int128_t)w*b / __gcd(w,b) > L ? L+1 : (__int128_t)w*b / __gcd(w,b);
    i64 k = L/l;
    i64 p = k*b + min(b, L%l+1) - 1;

    i64 q = L, d = __gcd(p, q);
    p /= d, q /= d;
    cout << p << "/" << q << "\n";
}
