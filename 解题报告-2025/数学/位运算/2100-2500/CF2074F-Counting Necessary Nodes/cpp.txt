using f64 = long double;

const f64 eps = 1e-12;
const i64 inf_i64 = 4e18;
const int inf_int = 1e9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    i64 l1, r1, l2, r2;
    cin >> l1 >> r1 >> l2 >> r2;

    i64 res = 0;
    for (int i = 0; i < 20; ++ i){
        i64 v = 1 << i+1;
        i64 nl1 = (l1+v-1)/v * v;
        i64 nr1 = r1/v * v;
        
        i64 nl2 = (l2+v-1)/v * v;
        i64 nr2 = r2 / v * v;

        if (nl1 > nr1 || nl2 > nr2){
            v /= 2;
            res += ((r1-l1)/v) * ((r2-l2)/v);
            break;
        } else {
            v /= 2;
            res += ((r1-l1)*(r2-l2) - (nr1-nl1)*(nr2-nl2)) / (v*v);
            l1 = nl1, r1 = nr1, l2 = nl2, r2 = nr2;
        }
    }

    cout << res << "\n";
}

    int _ = 1;
