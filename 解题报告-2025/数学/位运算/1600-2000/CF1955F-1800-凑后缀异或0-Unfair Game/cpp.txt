 把所有删去的数字放成一个数列，则每次删掉一个数时，剩余的是后缀。我们要最大化后缀异或和为 0 的个数。
我们就相当于把题目中的数排列好，使得前缀和异或为 0 的位置尽可能多。

#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
using f64 = long double;
using u64 = unsigned long long;
const f64 eps = 1e-12;
const i64 inf_i64 = 4e18;
const int inf_int = 1e9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int a, b, c, d;
    cin >> a >> b >> c >> d;
    cout << (a/2+b/2+c/2+d/2 + (a&b&c&1)) << "\n";
}
