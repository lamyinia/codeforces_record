const f64 eps = 1e-12;
const i64 inf_i64 = 4e18;
const int inf_int = 2e9;
template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i];
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int n, m;
    f64 p1;
    cin >> n >> p1 >> m;
    const f64 p2 = 1-p1;

    vector<vector<f64>> f(m+1, vector<f64>(n+1));
    
    f[0][0] = 1;
    for (int i = 0; i < m; ++ i){
        for (int j = 0; j <= n; ++ j){
            f64 &v = f[i][j];
            if (j == n){
                f[i+1][j] += v;
            } else {
                f[i+1][j+1] += p1 * v;
                f[i+1][j] += p2 * v;      
            }
        }
    }

    f64 res = 0;
    for (int j = 1; j <= n; ++ j){
        res += f[m][j] * j;
    }

    cout << fixed << setprecision(10) << res << "\n";
}
