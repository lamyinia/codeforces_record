template <class T> void fmax(T &a, const T &b){
    if (b > a) a = b;
}
template <class T> void fmin(T &a, const T &b){
    if (b < a) a = b;
}
template <class T> void print(vector <T> &A, int d = 0){
    int n = A.size();
    for (int i = 0; i < n; ++ i){
        if (i) cout << " ";
        cout << A[i]+d;
    }
    cout << "\n";
}
int jgs(f64 x){
    return x < -eps ? -1 : x > eps;
}

void solve(){
    int q;
    cin >> q;
    while (q--){
        i64 n, m, k;
        cin >> n >> m >> k;
        i64 res = 0;
        if (k < max(n, m)) res = -1;
        else if ((n+m)%2) res = k-1;
        else if (n%2 == k%2) res = k;
        else res = k-2;
        cout << res << "\n";
    }
}
